    <script>
        const csvInput = document.getElementById('csvInput');
        const welcomeScreen = document.getElementById('welcomeScreen');
        const dashboard = document.getElementById('dashboard');
        
        let chartInstances = {};

        csvInput.addEventListener('change', (e) => {
            const file = e.target.files[0];
            if (!file) return;
            Papa.parse(file, {
                header: true,
                delimiter: ";",
                skipEmptyLines: true,
                complete: function(results) { processData(results.data); }
            });
        });

        function categorize(text) {
            if (!text) return 'Outros';
            text = text.toLowerCase();
            if (/\b(cor|cores|sortid|estamp|azul|rosa|verde|branco|preto|cinza|bege|liso)\b/.test(text)) return 'Cores';
            if (/\b(tamanho|medida|dimens|cm|metro|solteiro|casal|queen|king|futon)\b/.test(text)) return 'Tamanho';
            if (/\b(envio|entrega|chega|prazo|rastre|postagem|demora|quando|transportadora)\b/.test(text)) return 'Envio';
            if (/\b(nota fiscal|nf|valor|paguei|preço|nota|fiscal)\b/.test(text)) return 'Financeiro/Nota';
            return 'Outros';
        }

        function processData(rawData) {
            const data = rawData.map(row => ({
                ...row,
                Category: categorize(row.Mensagem),
                Timestamp: new Date(row['Data da mensagem'])
            })).sort((a, b) => a.Timestamp - b.Timestamp);

            const conversations = {};
            data.forEach(msg => {
                const key = `${msg['Id do Pedido']}_${msg['SKU']}`;
                if (!conversations[key]) conversations[key] = [];
                conversations[key].push(msg);
            });

            let totalResponses = 0;
            let humanResponses = 0;
            let botResponses = 0;
            const conflicts = [];
            const missed = [];
            const catStats = { Cores: 0, Tamanho: 0, Envio: 0, 'Financeiro/Nota': 0, Outros: 0 };
            const skuHumanStats = {};

            Object.values(conversations).forEach(conv => {
                const botCats = new Set();
                const humanCats = new Set();

                for (let i = 0; i < conv.length; i++) {
                    const msg = conv[i];
                    
                    if (msg.Origem === 'BOT' || msg.Origem === 'ATTENDANT') {
                        totalResponses++;
                        if (msg.Origem === 'BOT') {
                            botResponses++;
                            botCats.add(msg.Category);
                        } else {
                            humanResponses++;
                            humanCats.add(msg.Category);
                            catStats[msg.Category] = (catStats[msg.Category] || 0) + 1;

                            const sku = msg.SKU;
                            if (!skuHumanStats[sku]) skuHumanStats[sku] = { name: msg.Itens, count: 0, themes: {} };
                            skuHumanStats[sku].count++;
                            skuHumanStats[sku].themes[msg.Category] = (skuHumanStats[sku].themes[msg.Category] || 0) + 1;
                        }
                    }

                    if (msg.Origem === 'CUSTOMER' && msg.Category !== 'Outros') {
                        const next = conv[i+1];
                        if (next && next.Origem === 'ATTENDANT') {
                            missed.push({
                                theme: msg.Category,
                                customer: msg.Mensagem,
                                human: next.Mensagem,
                                id: msg['Id do Pedido']
                            });
                        }
                    }
                }

                const overlap = [...botCats].filter(c => humanCats.has(c) && c !== 'Outros');
                overlap.forEach(cat => {
                    const bMsg = conv.find(m => m.Origem === 'BOT' && m.Category === cat);
                    const hMsg = conv.find(m => m.Origem === 'ATTENDANT' && m.Category === cat);
                    conflicts.push({ id: conv[0]['Id do Pedido'], sku: conv[0]['SKU'], theme: cat, bot: bMsg.Mensagem, human: hMsg.Mensagem });
                });
            });

            updateUI({
                total: totalResponses,
                human: humanResponses,
                bot: botResponses,
                conflicts: conflicts,
                missed: missed,
                catStats: catStats,
                skuHumanStats: skuHumanStats
            });
        }

        function updateUI(stats) {
            welcomeScreen.classList.add('hidden');
            dashboard.classList.remove('hidden');

            document.getElementById('kpiTotal').innerText = stats.total;
            document.getElementById('kpiHuman').innerText = stats.human;
            document.getElementById('kpiHumanPct').innerText = `${((stats.human / stats.total) * 100).toFixed(1)}% do total`;
            document.getElementById('kpiConflicts').innerText = stats.conflicts.length;
            document.getElementById('kpiMissed').innerText = stats.missed.length;

            // Calculo de ganhos
            const gainAutoPct = ((stats.missed.length / stats.total) * 100).toFixed(1);
            const timeSaved = (stats.human * 0.08).toFixed(1); // Média de 5 min por atendimento humano
            
            document.getElementById('gainAuto').innerText = gainAutoPct;
            document.getElementById('gainTime').innerText = timeSaved;

            document.getElementById('missedTable').innerHTML = stats.missed.slice(0, 8).map(m => `
                <tr>
                    <td class="px-6 py-4 font-bold text-amber-700">${m.theme}</td>
                    <td class="px-6 py-4 text-slate-600">"${m.customer.substring(0, 60)}..."</td>
                    <td class="px-6 py-4 font-medium text-emerald-700 italic">"${m.human.substring(0, 60)}..."</td>
                    <td class="px-6 py-4 font-mono text-slate-400">#${m.id}</td>
                </tr>`).join('');

            document.getElementById('conflictTable').innerHTML = stats.conflicts.slice(0, 8).map(c => `
                <tr>
                    <td class="px-6 py-4"><div class="font-bold text-blue-600">#${c.id}</div><div class="text-[10px] text-slate-400">${c.sku}</div></td>
                    <td class="px-6 py-4"><span class="bg-slate-100 px-2 py-1 rounded text-[10px] font-bold">${c.theme}</span></td>
                    <td class="px-6 py-4 text-slate-500 italic">"${c.bot.substring(0, 50)}..."</td>
                    <td class="px-6 py-4 font-bold">"${c.human.substring(0, 50)}..."</td>
                </tr>`).join('');

            renderCharts(stats);
            renderInsights(stats);
            renderTopSkus(stats.skuHumanStats);
        }

        function renderTopSkus(skuStats) {
            const list = document.getElementById('topSkuList');
            const sorted = Object.entries(skuStats).sort((a, b) => b[1].count - a[1].count).slice(0, 5);
            list.innerHTML = sorted.map(([sku, data]) => {
                const themeEntries = Object.entries(data.themes);
                const topTheme = themeEntries.length > 0 ? themeEntries.sort((a,b) => b[1]-a[1])[0][0] : 'N/A';
                return `
                <div class="p-3 bg-slate-50 rounded-xl border border-slate-100">
                    <div class="flex justify-between items-start mb-1">
                        <span class="text-[10px] font-mono text-blue-500 font-bold">${sku}</span>
                        <span class="bg-blue-100 text-blue-700 text-[10px] font-bold px-2 py-0.5 rounded-full">${data.count} atend.</span>
                    </div>
                    <div class="text-[11px] font-bold text-slate-800 line-clamp-1 mb-1">${data.name || 'Produto sem nome'}</div>
                    <div class="text-[10px] text-slate-500 flex items-center gap-1">
                        Principal tema: <b>${topTheme}</b>
                    </div>
                </div>`;
            }).join('');
        }

        function renderCharts(stats) {
            if (chartInstances.cat) chartInstances.cat.destroy();
            if (chartInstances.origin) chartInstances.origin.destroy();
            const ctxCat = document.getElementById('categoryChart').getContext('2d');
            chartInstances.cat = new Chart(ctxCat, {
                type: 'bar',
                data: { labels: Object.keys(stats.catStats), datasets: [{ data: Object.values(stats.catStats), backgroundColor: '#3b82f6', borderRadius: 8 }] },
                options: { maintainAspectRatio: false, plugins: { legend: { display: false } } }
            });
            const ctxOrigin = document.getElementById('originChart').getContext('2d');
            chartInstances.origin = new Chart(ctxOrigin, {
                type: 'doughnut',
                data: { labels: ['Bot', 'Humano'], datasets: [{ data: [stats.bot, stats.human], backgroundColor: ['#6366f1', '#10b981'], borderWidth: 0 }] },
                options: { maintainAspectRatio: false, cutout: '75%', plugins: { legend: { position: 'bottom' } } }
            });
        }

        function renderInsights(stats) {
            const list = document.getElementById('insightsList');
            const sortedMissed = stats.missed.reduce((acc, curr) => { acc[curr.theme] = (acc[curr.theme] || 0) + 1; return acc; }, {});
            const topMissed = Object.entries(sortedMissed).sort((a,b) => b[1] - a[1])[0];
            const sortedSkus = Object.entries(stats.skuHumanStats).sort((a, b) => b[1].count - a[1].count);
            const topSku = sortedSkus[0];

            list.innerHTML = `
                <li class="flex items-start gap-3 p-3 bg-slate-50 rounded-xl border-l-4 border-l-blue-600">
                    <span class="text-blue-600 mt-1">🚀</span>
                    <span><b>Foco no Produto:</b> O <span class="font-bold">${topSku ? topSku[1].name.substring(0,35) : 'N/A'}...</span> gera mais trabalho manual (${topSku ? topSku[1].count : 0} casos). Melhore a FAQ deste anúncio.</span>
                </li>
                <li class="flex items-start gap-3 p-3 bg-slate-50 rounded-xl border-l-4 border-l-amber-500">
                    <span class="text-amber-500 mt-1">⚠️</span>
                    <span><b>Gargalo:</b> O tema <b>${topMissed ? topMissed[0] : 'N/A'}</b> tem o maior volume de falhas de automação. Treine a IA para reconhecer pedidos de "não enviar cor X".</span>
                </li>
